------afficher le nombre de doublon

SELECT *
FROM transaction
WHERE account_id IN (
  SELECT account_id
  FROM transaction
  GROUP BY account_id
  HAVING COUNT(*) > 1
);

-----------------------------------jointure------------------------
Une jointure sert à relier des informations qui sont dans plusieurs tables.
Parce qu’en base de données on ne met pas tout dans une seule table.


-----afficher la balance et la description d un compte
select a.balance,transaction.description
from account a
join transaction on a.id=transaction.id;


----Lister les clients ayant plus d’un compte.
SELECT
    c.first_name,
    c.last_name,
    COUNT(a.id) AS nb_accounts
FROM Customer c
         JOIN Account a ON a.customer_id = c.id
GROUP BY c.id, c.first_name, c.last_name
HAVING COUNT(a.id) > 1;
